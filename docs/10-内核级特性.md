## 内联汇编 `asm!`

在 `unsafe {}` 块内使用，支持 GCC 扩展汇编语法。

### 语法

```shux
unsafe {
  asm!("template" : outputs : inputs : clobbers : options);
}
```

各段以 `:` 分隔，空段可省略尾部。`%` 在有操作数的 asm 中须写 `%%`（如 `%%eax`）。

### 操作数

```shux
// 输出操作数："=r"(var)
// 输入操作数："r"(expr), "a"(val), "d"(port)
// 读写操作数："+a"(var)
// 符号操作数："sym"(global_var)
// clobber："memory", "cc", "eax"
```

### options

| 选项 | 效果 |
|------|------|
| `pure` | 允许编译器消除/重排（无副作用时） |
| `nomem` | asm 不访问内存（可优化） |
| `readonly` | asm 只读内存（可优化） |
| `nostack` | asm 不使用栈 |
| `preserves_flags` | asm 不修改标志位 |
| `noreturn` | asm 不返回 |

**默认**：`__asm__ __volatile__`（不可消除，内核 MMIO 安全）。`pure`/`nomem`/`readonly` 时改为 `__asm__`（可优化）。

### 示例

```shux
// 写串口
unsafe { asm!("outb %%al, %%dx" : : "a"(c), "d"(0x3F8)); }

// 读 CR0
let cr0: u32 = 0;
unsafe { asm!("mov %%cr0, %0" : "=a"(cr0)); }

// 原子加
unsafe { asm!("lock xadd %0, (%1)" : "+a"(val) : "r"(addr) : "memory", "cc"); }

// 符号引用
unsafe { asm!("mov %1, %0" : "=r"(val) : "sym"(global_var)); }
```

## `*volatile T` — 语言级 volatile

指针类型限定符，确保 MMIO 读写不被编译器消除。

```shux
let mmio: *volatile u32 = 0xB8000 as *volatile u32;
unsafe { *mmio = 0x0741; }      // 写不被 DCE
let val: u32 = 0;
unsafe { val = *mmio; }          // 探测读保留
```

## 内核属性

所有属性以 `#[attr]` 形式写在 `function`/`struct`/`let`/`const` 前面。

### `#[naked]` — 无 prologue/epilogue 函数

函数体仅允许 `asm!`，编译器不生成 `push %ebp; mov %esp, %ebp` 等栈帧代码。

```shux
#[naked]
function switch_to(old: *Context, new_ctx: *Context): void {
  unsafe {
    asm!("movl 4(%esp), %eax; ...; ret");
  }
}
```

### `#[entry]` — 内核入口

将函数改名为 `_start`，标记为 naked + noreturn + 外部链接，供 bootloader 跳转。

```shux
#[entry]
function start(): void {
  // 清 .bss + 设栈 + 调 kmain + 关中断 + 停机
  unsafe { asm!("cld; mov $__bss_start, %edi; mov $__bss_end, %ecx; sub %edi, %ecx; xor %eax, %eax; rep stosb; mov $__stack_top, %esp; call kmain; cli; hlt"); }
}
```

### `#[used]` — 不被编译器消除

外部链接 + `__attribute__((used))`，防止仅被 asm 引用的函数被 DCE。

```shux
#[used]
function timer_handler(): void { ... }
```

### `#[interrupt]` — 中断处理函数

C 编译器自动保存/恢复寄存器 + `iret` 返回。x86 上等价于 `__attribute__((interrupt))`。

```shux
#[interrupt]
function timer_handler(): void {
  tick_count = tick_count + 1;
  unsafe { asm!("movb $0x20, %al; outb %al, $0x20"); }  // EOI
}
```

### `#[link_section("name")]` — 指定段

将 `const`/`let`/`function` 放入指定 ELF 段。

```shux
#[link_section(".boot")]
const mb1: MB1Header = { magic: 0x1BADB002, ... };
```

### `#[no_mangle]` — 外部链接不 DCE

函数名不修改，外部链接，不被 DCE。

```shux
#[no_mangle]
function exported_func(): i32 { return 42; }
```

### `#[link_name("name")]` — 指定外部符号名

`extern function` 使用指定符号名链接。

```shux
#[link_name("asm_multiply")]
extern function asm_mul(a: i32, b: i32): i32;
```

### `#[max_stack(N)]` — 栈用量上限

标记函数栈帧上限，供编译期分析。

```shux
#[max_stack(64)]
function small_func(a: i32, b: i32): i32 { ... }
```

### `#[global_allocator]` — 全局分配器入口

标记函数为全局分配器，外部链接 + 不 DCE。

```shux
#[global_allocator]
function kalloc(size: u32): u32 { ... }
```

### `#[cold]` — 冷路径优化提示

```shux
#[cold]
function error_handler(code: i32): void { ... }
```

### `#[inline(never)]` / `#[inline(always)]` — 内联控制

```shux
#[inline(never)]
function big_function(): i32 { ... }

#[inline(always)]
function fast_add(a: i32, b: i32): i32 { return a + b; }
```

### `#[export_name("name")]` — 导出符号名

```shux
#[export_name("my_custom_name")]
function original_name(): i32 { return 42; }
// → int32_t my_custom_name() { ... }
```

### `#[panic_handler]` — panic 处理函数

```shux
#[panic_handler]
function kernel_panic(msg: i32): void {
  // 串口输出 + halt
  unsafe { asm!("cli; hlt"); }
}
```

### `#[thread_local]` — 线程局部存储

```shux
#[thread_local]
let per_thread_counter: i32 = 0;
```

→ `__thread static int32_t per_thread_counter = 0;`

### `#[percpu]` — per-CPU 数据段

```shux
#[percpu]
let cpu_local_data: i32 = 0;
```

→ `__attribute__((section(".percpu"))) static int32_t cpu_local_data = 0;`

### `#[send]` / `#[sync]` — 线程安全契约

标记 struct 可安全跨线程传递 (`#[send]`) 或共享 (`#[sync]`)。

```shux
#[send]
#[sync]
struct SafeBuffer { data: *u8; len: u32; }
```

## 编译器修复

| 修复 | 影响 |
|------|------|
| 前向声明 | `#[used]`/`#[interrupt]`/`#[naked]`/`#[entry]` 函数的外部前向声明 |
| 括号丢失 | `(a & 1) == 0` 不再被错编为 `a & 1 == 0` |
| 位运算括号 | `(a \| b) & c` 不再被错编为 `a \| b & c` |
| K8 .data 初始化 | 顶层 `let` 初值落 `.data`（内核 `_start` 不调 `init_globals`） |

## 原子操作

### 底层 asm 原子（已实现）

```shux
// atomic fetch_add
function atomic_fetch_add_u32(addr: *volatile u32, val: u32): u32 {
  unsafe { asm!("lock xadd %0, (%1)" : "+a"(val) : "r"(addr) : "memory", "cc"); }
  return val;
}

// atomic compare-and-swap
function atomic_cas_u32(addr: *volatile u32, expected: u32, newval: u32): u32 {
  let prev: u32 = expected;
  unsafe { asm!("lock cmpxchg %2, (%1)" : "+a"(prev) : "r"(addr), "r"(newval) : "memory", "cc"); }
  return prev;
}

// 自旋锁
function spinlock_acquire(addr: *volatile u32): void {
  let old: u32 = 1;
  while (old != 0) {
    old = 1;
    unsafe { asm!("xchg %0, (%1)" : "+a"(old) : "r"(addr) : "memory"); }
  }
}
```

### 高层原子类型（已实现）

`AtomicU32` 封装类型提供 load/store/fetch_add/cas 接口（见 [intrinsics.x](../tests/kernel/intrinsics.x)）。

> **状态**：底层 asm 原子 ✅ 已实现（QEMU 验证） | 高层 AtomicU32 ✅ 已实现（intrinsics.x QEMU gate）

## `unsafe` 规则

### 必须在 `unsafe {}` 块内使用的操作

| 操作 | 原因 |
|------|------|
| `asm!(...)` | 直接操作硬件寄存器/内存 |
| `*volatile T` 解引用 | MMIO 读写，编译器不可优化 |
| `*T` 解引用 | 裸指针无安全检查 |
| 整数→指针强转 `0xB8000 as *u8` | 创建任意地址指针 |
| `extern function` 调用（非 C ABI 安全时） | 外部代码可能不安全 |

### `unsafe` 块作用域

```shux
function kmain(): i32 {
  // unsafe 块内的操作只在块内有效
  unsafe {
    asm!("sti");
    let p: *volatile u32 = 0xB8000 as *volatile u32;
    *p = 0x0741;
  }
  // 块外不能使用 unsafe 操作
  return 0;
}
```

### 内核全局 unsafe 假设

内核运行在 ring 0，所有硬件操作都在 `unsafe` 内。`#[naked]` 函数体只能包含 `asm!`。`#[entry]` 函数是内核入口，整个函数体在裸机环境执行。

> **状态**：✅ 已实现 — `asm!` 和裸指针解引用强制 `unsafe`，typeck 检查

## Boot 与链接

### Multiboot1（已实现）

```shux
#[repr(C)]
struct MB1Header {
  magic: u32;
  flags: u32;
  checksum: u32;
}

#[link_section(".boot")]
const mb1: MB1Header = {
  magic: 0x1BADB002,
  flags: 0,
  checksum: 0xE4524FFE,  // 0x100000000 - (magic + flags)
};
```

### Multiboot2（已实现）

Multiboot2 头含 end tag（type=0/flags=0/size=8），header_length=24，4 字节对齐。QEMU `-kernel` 仅支持 multiboot1，multiboot2 需 GRUB。Gate: `multiboot2-gate.sh` 7/7 PASS。

### UEFI 应用（已实现）

PE32+ 格式 EFI 应用骨架，含 `EFI_SYSTEM_TABLE` 协议头和 `efi_main` 入口。`zig cc -target x86_64-windows-gnu` 编译为 PE32+。Gate: `uefi-app-gate.sh` 6/6 PASS。

### Linker Script

```ld
/* kernel.ld — 32 位 */
ENTRY(_start)
SECTIONS {
    . = 0x100000;
    .boot : { *(.boot) }     /* multiboot 头必须在最前 */
    .text : { *(.text) }
    .rodata : { *(.rodata*) }
    .data : { *(.data) }
    .bss : {
        __bss_start = .;     /* _start 用 rep stosb 清零 */
        *(.bss)
        *(COMMON)
        __bss_end = .;
        . = ALIGN(16);
        . += 16384;           /* 16KB 内核栈 */
        __stack_top = .;
    }
}
```

### `.bss` 清零与 `init_globals`

- **`.bss`**：零初始化的全局变量，bootloader 加载时自动清零
- **`.data`**：有初值的全局变量，bootloader 加载时自动写入（K8 修复确保 shux 顶层 `let` 初值落 `.data`）
- **`init_globals()`**：普通程序中 `main()` 调用此函数设置 `let` 初值；内核 `_start` 不调用，依赖 `.data` 段

> **注意**：shux 顶层 `let` 的初值如果是 C 常量表达式（整数字面量/已折叠常量），K8 修复后直接落 `.data`，内核可用。非恒量初值（如函数调用）仍走 `init_globals()`，内核不可用。

## 常见陷阱与 Pitfalls

### 1. naked 函数体只能有 `asm!`

`#[naked]` 函数不能有局部变量、C 语句，只能包含 `asm!` 表达式。否则编译器会生成 prologue/epilogue。

```shux
// ✅ 正确
#[naked]
function switch_to(old: *Context, new_ctx: *Context): void {
  unsafe { asm!("movl 4(%esp), %eax; ...; ret"); }
}

// ❌ 错误 — naked 函数不能有局部变量
#[naked]
function bad(): void {
  let x: i32 = 0;  // 编译器会生成栈帧
  unsafe { asm!("nop"); }
}
```

### 2. `#[used]` 防止 DCE

仅被 `asm!("mov $func_name, %0")` 引用的函数会被 DCE 消除。必须加 `#[used]`。

```shux
// ❌ 不加 #[used] — 函数被 DCE，asm! 引用变为未定义符号
function timer_handler(): void { ... }

// ✅ 正确
#[used]
function timer_handler(): void { ... }
```

### 3. `%%` vs `%` in asm!

有操作数（outputs/inputs/clobbers 段存在）时，寄存器名必须用 `%%`：

```shux
// 有操作数 → 用 %%eax
unsafe { asm!("outb %%al, %%dx" : : "a"(c), "d"(0x3F8)); }

// 无操作数 → 用 %eax
unsafe { asm!("mov $0x80000, %esp; call kmain; cli; hlt"); }
```

### 4. `#[interrupt]` 函数需 `(void)` 参数

x86 `__attribute__((interrupt))` 要求至少一个指针参数。shux codegen 自动添加 `void *__shux_isr_frame`。

### 5. 顶层 `let` 初始化与内核

内核 `_start` 不调用 `init_globals()`。顶层 `let` 的初值必须是 C 常量（整数字面量/已折叠常量），否则内核中变量值为 0。K8 修复后，C 常量初值直接落 `.data` 段。

### 6. 比较运算符括号

`(a & 1) == 0` 在 codegen 中曾被错编为 `a & 1 == 0`（C 优先级 bug）。已修复（v3 编译器修复），但建议复杂表达式拆分到 `let` 变量。

### 7. `#[used]`/`#[interrupt]` 函数的 forward declaration

这些函数的外部前向声明必须是 external（非 `static inline`），否则被调用的 helper 函数会被 DCE。已修复。

## `asm goto!` — 带标签跳转的内联汇编

GCC `asm goto` 语法，允许汇编模板跳转到 C 标签。用于条件分支、错误处理。

### 语法

```shux
unsafe {
  asm goto!("jmp %l[skip]" : : : "memory" : skip);
}
// codegen 自动在 asm 语句后生成 C 标签 skip:
```

第五段（`:` 后）为 label 标识符列表，对应 GCC `asm goto` 的 label section。codegen 发射 `__asm__ goto(...)` 并自动在语句后生成 C 标签定义。

### 示例

```shux
unsafe {
  asm goto!("jne %l[error]" : : : : error);
}
// 正常路径
return 0;
// error 标签由 codegen 自动生成
```

## 位域 (bitfield)

C 风格位域，用于紧凑的硬件寄存器描述（页表项、IDT 项等）。

### 语法

```shux
#[repr(C)]
struct PageFlags {
  present: u32 : 1;
  writable: u32 : 1;
  user: u32 : 1;
  addr: u32 : 20;
}
```

字段类型后 `: N` 指定位宽。codegen 发射 C 位域 `uint32_t present : 1;`。

## AtomicU32 — 高层原子类型

封装 `asm!` 原子操作的 struct 类型，提供 load/store/fetch_add/cas 接口。

```shux
#[repr(C)]
struct AtomicU32 { value: u32; }

function atomic_u32_load(a: *AtomicU32): u32 { ... }
function atomic_u32_store(a: *AtomicU32, val: u32): void { ... }
function atomic_u32_fetch_add(a: *AtomicU32, val: u32): u32 { ... }
function atomic_u32_cas(a: *AtomicU32, expected: u32, newval: u32): u32 { ... }
```

## 高层 intrinsic 函数

常用 x86 指令的封装函数，避免直接写 `asm!`：

```shux
intrinsic_hlt()      // hlt
intrinsic_cli()      // cli
intrinsic_sti()      // sti
intrinsic_pause()    // pause (spinlock 优化)
intrinsic_mfence()   // mfence (全屏障)
intrinsic_lfence()   // lfence (读屏障)
intrinsic_sfence()   // sfence (写屏障)
intrinsic_read_cr0() // mov %cr0
intrinsic_read_cr3() // mov %cr3
intrinsic_cpuid(leaf) // cpuid
intrinsic_rdmsr(msr)  // rdmsr
intrinsic_wrmsr(msr, val) // wrmsr
```

## Multiboot2 头

Multiboot2 引导协议头（含 end tag），用于 GRUB 引导。QEMU `-kernel` 仅支持 multiboot1，multiboot2 需 GRUB。

```shux
#[repr(C)]
struct MB2Header {
  magic: u32;          // 0xE85250D6
  arch: u32;           // 0 = i386
  header_length: u32;  // 24
  checksum: u32;       // -(magic + arch + header_length)
  end_tag_tf: u32;     // 0 (type=0 + flags=0)
  end_tag_size: u32;   // 8
}

#[link_section(".boot")]
const mb2: MB2Header = { magic: 0xE85250D6, arch: 0, header_length: 24, checksum: 0x17ADAF12, end_tag_tf: 0, end_tag_size: 8 };
```

## x86_64 长模式启动

32 位 multiboot1 入口 → 长模式切换 → 64 位代码执行。

```
longmode_boot.s (32-bit) → _start: 设置页表/PAE/EFER/CR0/GDT → ljmp $0x08, _start64
                         → _start64 (64-bit): 串口输出 → hlt
longmode.x (shux): multiboot1 头 + kmain
```

QEMU gate: `make kernel-longmode`

## UEFI 应用

PE32+ 格式 EFI 应用骨架，含 EFI_SYSTEM_TABLE 协议头和 `efi_main` 入口。

```shux
#[repr(C)]
struct EFI_SYSTEM_TABLE {
  hdr: u64; firmware_vendor: u64; firmware_revision: u32;
  con_in: u64; con_out: u64; boot_services: u64; ...
}

#[used] #[no_mangle]
function efi_main(image_handle: u64, system_table: u64): u64 {
  // UEFI 应用逻辑
  return 0;  // EFI_SUCCESS
}
```

编译: `zig cc -target x86_64-windows-gnu` (PE32+ 输出)

## IST (Interrupt Stack Table)

x86_64 TSS + 64 位 IDT 门描述符，含 IST 字段用于中断栈切换（双错误、NMI 等）。

```shux
#[repr(C)]
struct TSS64 {
  reserved0: u32;
  rsp0_low: u32; rsp0_high: u32;
  // ... rsp1, rsp2 ...
  ist1_low: u32; ist1_high: u32;  // IST1 栈指针
  // ... ist2-ist7 ...
  iopb_offset: u32;
}

#[repr(C)]
struct IDTGate64 {
  offset_low: u32; selector: u32;
  ist: u32;          // IST 索引 (0=禁用, 1-7=使用 TSS.istN)
  type_attr: u32;    // 0x8E = 中断门, P+DPL=0
  offset_mid: u32; offset_high: u32; reserved: u32;
}
```

## 与 C 的互操作性

### `extern function` — 声明 C 函数

```shux
extern function memcpy(dst: *u8, src: *u8, n: u32): *u8;
```

### 手写 `.s` 汇编与 shux 互链

```shux
// shux 侧
#[link_name("asm_multiply")]
extern function asm_mul(a: i32, b: i32): i32;
```

```asm
// asm_multiply.s
.globl asm_multiply
asm_multiply:
    movl 4(%esp), %eax
    movl 8(%esp), %ecx
    imull %ecx, %eax
    ret
```

### ABI 兼容性

shux C 前端生成标准 C 代码，ABI 与 C 完全兼容：
- 函数名不 mangling（`#[no_mangle]` 进一步确保）
- 参数传递遵循 C ABI（cdecl / 当前 target 默认）
- 结构体布局遵循 `#[repr(C)]`
- `extern function` 直接映射到 C `extern` 声明

## 内核开发快速上手

### 步骤 1：最小内核

```shux
#[repr(C)]
struct MB1Header { magic: u32; flags: u32; checksum: u32; }

#[link_section(".boot")]
const mb1: MB1Header = {
  magic: 0x1BADB002,
  flags: 0,
  checksum: 0xE4524FFE,
};

function kmain(): i32 {
  // 在这里写内核逻辑
  return 0;
}

#[entry]
function start(): void {
  // 清 .bss + 设栈 + 调 kmain + 关中断 + 停机
  unsafe { asm!("cld; mov $__bss_start, %edi; mov $__bss_end, %ecx; sub %edi, %ecx; xor %eax, %eax; rep stosb; mov $__stack_top, %esp; call kmain; cli; hlt"); }
}

function main(): i32 { return kmain() + mb1.magic as i32; }
```

```sh
sh tests/kernel/build-kernel.sh my_kernel.x kernel.elf
qemu-system-x86_64 -kernel kernel.elf -no-reboot -display none -serial file:serial.log
cat serial.log
```

### 步骤 3：串口输出

```shux
function serial_putc(c: u8): void {
  unsafe { asm!("outb %%al, %%dx" : : "a"(c), "d"(0x3F8)); }
}

function kmain(): i32 {

}
```

### 步骤 4：中断

```shux

}
```
